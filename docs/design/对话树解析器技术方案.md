## 🎯 功能目标

实现一个**对话树解析器**，负责：
1. 加载和解析 JSON 格式的对话树文件
2. 根据触发条件筛选可用对话节点
3. 管理对话状态和分支选择
4. 与 DialogueManager 集成，提供对话数据

```gdscript
class_name DialogueTrigger
extends Resource

@export var day_min: int = 1
@export var day_max: int = 10
@export var trust_min: int = 0
@export var trust_max: int = 100
@export var event_flags_required: Array[String]
@export var event_flags_forbidden: Array[String]
@export var items_required: Array[String]


## 检查条件是否满足
func check_conditions(game_state: GameState) -> bool:
	var day = game_state.current_day
	var trust = game_state.get_trust(npc_id)
	
	if day < day_min or day > day_max:
		return false
	
	if trust < trust_min or trust > trust_max:
		return false
	
	for flag in event_flags_required:
		if not game_state.has_flag(flag):
			return false
	
	for flag in event_flags_forbidden:
		if game_state.has_flag(flag):
			return false
	
	return true
```

## 🔧 核心功能实现

### 1. 加载对话树

```gdscript
# dialogue_tree_parser.gd

func load_npc_tree(npc_id: String) -> DialogueTree:
	# 检查缓存
	if _loaded_trees.has(npc_id):
		return _loaded_trees[npc_id]
	
	# 加载 JSON 文件
	var file_path = "res://resources/data/dialogue/%s.json" % npc_id
	var file = FileAccess.open(file_path, FileAccess.READ)
	
	if not file:
		push_error("Failed to load dialogue tree: %s" % file_path)
		return null
	
	var json_text = file.get_as_text()
	file.close()
	
	# 解析 JSON
	var json = JSON.new()
	var parse_result = json.parse(json_text)
	
	if parse_result != OK:
		push_error("JSON parse error: %s" % json.get_error_message())
		return null
	
	# 创建 DialogueTree 对象
	var tree_data = json.data
	var tree = DialogueTree.new()
	tree.npc_id = tree_data.npc_id
	tree.npc_name = tree_data.npc_name
	tree.version = tree_data.version
	
	# 解析信任阶段
	for stage_data in tree_data.trust_stages.values():
		var stage = TrustStage.new()
		stage.name = stage_data.name
		stage.range_min = stage_data.range[0]
		stage.range_max = stage_data.range[1]
		tree.trust_stages.append(stage)
	
	# 解析对话节点
	for node_data in tree_data.nodes:
		var node = _parse_node(node_data)
		tree.nodes.append(node)
	
	# 缓存
	_loaded_trees[npc_id] = tree
	_active_tree = tree
	
	return tree


func _parse_node(node_data: Dictionary) -> DialogueNode:
	var node = DialogueNode.new()
	node.node_id = node_data.node_id
	
	# 解析触发条件
	node.trigger = _parse_trigger(node_data.trigger)
	
	# 解析对话行
	for line_data in node_data.dialogue:
		var line = DialogueLine.new()
		line.speaker = line_data.speaker
		line.text = line_data.text
		line.emotion = line_data.get("emotion", "neutral")
		node.dialogue.append(line)
	
	# 解析选项
	for choice_data in node_data.choices:
		var choice = DialogueChoice.new()
		choice.id = choice_data.id
		choice.text = choice_data.text
		choice.trust_change = choice_data.trust_change
		choice.next_node = choice_data.next_node
		node.choices.append(choice)
	
	return node
```

### 2. 获取可用节点

```gdscript
# dialogue_tree_parser.gd

func get_available_nodes(tree: DialogueTree) -> Array[DialogueNode]:
	var available: Array[DialogueNode] = []
	
	# 获取当前游戏状态
	var game_state = GameManager.get_game_state()
	var trust = TrustManager.get_trust(tree.npc_id)
	
	# 遍历所有节点，检查触发条件
	for node in tree.nodes:
		if node.can_trigger(game_state):
			available.append(node)
	
	# 按优先级排序（特殊节点优先）
	available.sort_custom(_sort_nodes_by_priority)
	
	return available


func _sort_nodes_by_priority(a: DialogueNode, b: DialogueNode) -> bool:
	# 特殊节点优先级最高
	if a.is_special and not b.is_special:
		return true
	if b.is_special and not a.is_special:
		return false
	
	# 否则按信任值范围排序（窄范围优先）
	var a_range = a.trigger.trust_max - a.trigger.trust_min
	var b_range = b.trigger.trust_max - b.trigger.trust_min
	
	return a_range < b_range
```

### 3. 处理对话选择

```gdscript
# dialogue_tree_parser.gd

func select_choice(choice_id: String) -> DialogueNode:
	if not _active_tree or not _current_node:
		push_error("No active dialogue")
		return null
	
	# 找到选择的选项
	var selected_choice: DialogueChoice = null
	
	for choice in _current_node.choices:
		if choice.id == choice_id:
			selected_choice = choice
			break
	
	if not selected_choice:
		push_error("Choice not found: %s" % choice_id)
		return null
	
	# 更新信任值
	if selected_choice.trust_change != 0:
		TrustManager.add_trust(_active_tree.npc_id, selected_choice.trust_change)
	
	# 获取下一个节点
	var next_node = _active_tree.get_node_by_id(selected_choice.next_node)
	
	if next_node:
		_current_node = next_node
	
	return next_node
```

### 4. 与 DialogueManager 集成

```gdscript
# dialogue_manager.gd

class_name DialogueManager
extends Node

@onready var parser = DialogueTreeParser.new()

## 当前对话 NPC
var current_npc_id: String

## 是否正在对话中
var is_in_dialogue: bool = false


## 开始与 NPC 的对话
func start_dialogue(npc_id: String) -> void:
	# 加载对话树
	var tree = parser.load_npc_tree(npc_id)
	
	if not tree:
		push_error("Failed to load dialogue tree for: %s" % npc_id)
		return
	
	current_npc_id = npc_id
	is_in_dialogue = true
	
	# 获取可用节点
	var available_nodes = parser.get_available_nodes(tree)
	
	if available_nodes.is_empty():
		# 没有可用对话，结束
		end_dialogue()
		return
	
	# 选择第一个可用节点（或根据优先级）
	parser._current_node = available_nodes[0]
	
	# 显示对话 UI
	_show_dialogue()


## 处理玩家选择
func handle_choice(choice_id: String) -> void:
	var next_node = parser.select_choice(choice_id)
	
	if next_node:
		_show_dialogue()
	else:
		end_dialogue()


## 结束对话
func end_dialogue() -> void:
	is_in_dialogue = false
	current_npc_id = ""
	parser.reset()
	
	# 隐藏对话 UI
	_hide_dialogue()


func _show_dialogue():
	# 调用对话 UI 显示
	var node = parser._current_node
	DialogueUI.show_dialogue(node.dialogue, node.choices)


func _hide_dialogue():
	DialogueUI.hide()
```

## 📝 技术备注

### 1. 缓存策略

- 对话树加载后缓存到内存
- 使用 Dictionary 以 npc_id 为键
- 游戏结束时清空缓存

### 2. 错误处理

- JSON 解析失败时返回 null 并记录错误
- 节点不存在时跳过，不影响其他节点
- 条件检查失败不报错，只是节点不可用

### 3. 扩展性

- 预留 LLM 接口（二周目模式）
- 支持动态加载新对话树（DLC）
- 支持多语言（对话文本本地化）
